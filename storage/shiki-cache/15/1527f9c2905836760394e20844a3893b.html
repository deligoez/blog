// Database returns integer 1, not boolean true
$user = User::find(1);
$user->is_active;  // Returns: 1 (integer)

// This passes—but is it really a boolean?
$this->assertTrue($user->is_active);  // 1 is "truthy" ✓

// This catches the missing boolean cast:
$this->assertSame(true, $user->is_active);  // 1 !== true ✗
