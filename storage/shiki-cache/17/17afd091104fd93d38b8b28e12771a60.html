trait CleansUpTestFiles
{
    protected array $testFilePaths = [];

    protected function tearDownCleansUpTestFiles(): void
    {
        foreach ($this->testFilePaths as $path) {
            if (file_exists($path)) {
                @unlink($path);
            }
        }
        $this->testFilePaths = [];
    }

    protected function trackTestFile(string $path): string
    {
        $this->testFilePaths[] = $path;
        return $path;
    }
}
