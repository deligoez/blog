// Option 1: Use parse() instead of createFromFormat()
$deadline = Carbon::parse('2025-01-15');  // 00:00:00

// Option 2: Explicitly set the time
$deadline = Carbon::createFromFormat('Y-m-d', '2025-01-15')
    ->startOfDay();

// Option 3: Include time in the format
$deadline = Carbon::createFromFormat('Y-m-d H:i:s', '2025-01-15 00:00:00');
