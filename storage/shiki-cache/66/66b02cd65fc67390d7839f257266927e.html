protected function actingAsMerchantUser(): self
{
    $retailer = Merchant::factory()->create();
    $user = MerchantUser::factory()
        ->for($retailer)
        ->owner()
        ->create();

    Passport::actingAs($user, [], 'retailer');

    return $this;
}

protected function actingAsCustomer(): self
{
    $farmer = Customer::factory()->create();
    Passport::actingAs($farmer, [], 'api_mobile');

    return $this;
}
