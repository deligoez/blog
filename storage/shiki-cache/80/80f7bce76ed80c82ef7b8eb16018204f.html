/**
 * Use this trait in tests that mock the Application container
 * and may cause Sushi model caching issues.
 */
trait RefreshesSushiModels
{
    protected function tearDownRefreshesSushiModels(): void
    {
        $reflection = new ReflectionClass(CachedProduct::class);
        $property = $reflection->getProperty('sushiConnection');
        $property->setAccessible(true);
        $property->setValue(null, null);
    }
}
